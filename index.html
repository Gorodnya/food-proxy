exports.handler = async (event) => { const { default: fetch } = await import('node-fetch'); try { // Ожидаемый путь: /proxy/bolt/food.bolt.eu/uk-ua/158-kyiv/search/?query=... const pathParts = event.path.split('/').filter(Boolean); // pathParts: ["proxy","bolt","food.bolt.eu","uk-ua","158-kyiv","search", ...] const host = pathParts[2]; // "food.bolt.eu" или "glovoapp.com" const targetPath = '/' + pathParts.slice(3).join('/'); // "/uk-ua/158-kyiv/search/..." const targetUrl = `https://${host}${targetPath}`; console.log('Proxy target:', targetUrl); const response = await fetch(targetUrl, { headers: { 'User-Agent': 'Mozilla/5.0 (iPhone; CPU iPhone OS 15_0 like Mac OS X)', 'Accept': 'text/html,*/*;q=0.9' } }); const headers = {}; response.headers.forEach((value, key) => { const k = key.toLowerCase(); if (!k.includes('x-frame-options') && !k.includes('content-security-policy')) { headers[key] = value; } }); const body = await response.text(); return { statusCode: response.status, headers: { ...headers, 'Content-Type': 'text/html; charset=utf-8' }, body }; } catch (error) { console.error('Proxy error:', error); return { statusCode: 500, headers: { 'Content-Type': 'text/html; charset=utf-8' }, body: `
Error: ${error.message}\nPath: ${event.path}
` }; } };
